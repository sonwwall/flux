pub async fn agent_reasoning_loop(ctx: &Context) -> Result<(), Error> {
    // 1. 感知层：获取当前环境快照
    let snapshot = capture_system_state().await?;

    loop {
        // 2. 推理层：计算最佳行动路径
        let action_plan = ctx.brain
            .analyze(&snapshot)
            .map_err(|e| Error::ReasoningFailed(e))?;

        if action_plan.is_terminal() {
            break;
        }

        // 3. 执行层：原子性应用变更
        let result = execute_step(action_plan).await?;

        // 4. 优化：记录指标并调整参数
        telemetry_log("step_complete", result);
    }

    Ok(())
}
